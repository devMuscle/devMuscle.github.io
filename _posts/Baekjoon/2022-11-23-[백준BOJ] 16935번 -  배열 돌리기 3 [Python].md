**문제 링크** : <https://www.acmicpc.net/problem/16953>

## 풀이 방법

문제의 요구사항대로 6가지의 함수를 만들어서 인덱스를 적절히 접근해서 구현을 하면 된다.

## 느낀 점

배열을 활용하는 문제여서 파이썬으로 편리하게 풀 수 있지 않을까 하는 생각을 했지만,
단순히 index로 접근하는 문제여서 C++에서 비해서 큰 편리함은 느끼지 못했다.

```python
import sys
input = sys.stdin.readline

def fun1() :
    new_arr = [[0 for _ in range(m)] for _ in range(n)]

    for i in range(len(new_arr)) :
        new_arr[i] = arr[n-i-1]

    return new_arr

def fun2() :
    new_arr = [[0 for _ in range(m)] for _ in range(n)]

    for i in range(n):
      for j in range(m):
        new_arr[i][j] = arr[i][m-1-j]

    return new_arr

def fun3() :
    new_arr = [[0 for _ in range(m)] for _ in range(n)]

    for i in range(len(new_arr)) :
        for j in range(len(new_arr[i])) :
            new_arr[i][j] = arr[n-j-1][i]

    return new_arr

def fun4() :
    new_arr = [[0 for _ in range(m)] for _ in range(n)]

    for i in range(len(new_arr)) :
        for j in range(len(new_arr[i])) :
            new_arr[i][j] = arr[j][n-i-1]

    return new_arr

def fun5():
    new_arr = [[0] * m for _ in range(n)]

    for i in range(n//2):
        for j in range(m//2):
            new_arr[i][j] = arr[i+n//2][j]

    for i in range(n//2, n):
        for j in range(m//2):
            new_arr[i][j] = arr[i][j+m//2]

    for i in range(n//2, n):
        for j in range(m//2, m):
            new_arr[i][j] = arr[i - n//2][j]

    for i in range(n//2):
        for j in range(m//2, m):
            new_arr[i][j] = arr[i][j-m//2]

    return new_arr

def fun6():
    new_arr = [[0] * m for _ in range(n)]

    for i in range(n//2):
        for j in range(m//2):
            new_arr[i][j] = arr[i][j-m//2]

    for i in range(n//2, n):
        for j in range(m//2):
            new_arr[i][j] = arr[i-n//2][j]

    for i in range(n//2, n):
        for j in range(m//2, m):
            new_arr[i][j] = arr[i][j - m//2]

    for i in range(n//2):
        for j in range(m//2, m):
            new_arr[i][j] = arr[i-n//2][j]

    return new_arr

n,m,r = map(int, input().split())

arr = [[0 for _ in range(m)] for _ in range(n)]

for i in range(n) :
   arr[i] = list(map(int, input().split()))

for _ in range(r) :
    cals = list(map(int, input().split()))

    for cal in cals :
        if cal == 1:
            arr = fun1()
        elif cal == 2 :
            arr = fun2()
        elif cal == 3 :
            arr = fun3()
            n = m
            m = n
        elif cal == 4 :
            arr = fun4()
            n = m
            m = n
        elif cal == 5 :
            arr = fun5()
        elif cal == 6 :
            arr = fun6()

for i in range(len(arr)) :
    print(*arr[i])
```
